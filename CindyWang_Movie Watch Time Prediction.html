In [19]:
from sklearn.model_selection import train_test_split
from sklearn import linear_model
from sklearn.metrics import mean_squared_error, r2_score
from math import sqrt

lr_train,lr_validate=train_test_split(model_train,test_size=0.2,random_state=0) #validation = 15% of train set

from sklearn.preprocessing import PolynomialFeatures
poly = PolynomialFeatures(2)

lr_train_x=lr_train.drop(['video_id','cvt_per_day'],axis=1)
lr_validate_x=lr_validate.drop(['video_id','cvt_per_day'],axis=1)
lr_train_y=lr_train['cvt_per_day']
lr_validate_y=lr_validate['cvt_per_day']
lr_train_xp = poly.fit_transform(lr_train_x)
lr_validate_xp = poly.fit_transform(lr_validate_x)

alphas=np.logspace(-2.6, 2.5,num=150)
scores=np.empty_like(alphas) 
opt_a=float('-inf')
max_score=float('-inf')
for i,a in enumerate(alphas):
    lasso=linear_model.Lasso()
    lasso.set_params(alpha=a)
    lasso.fit(lr_train_xp,lr_train_y)
    scores[i]=lasso.score(lr_validate_xp,lr_validate_y)
    if scores[i]>max_score:
        opt_a=a
        max_score=scores[i]
        lasso_save=lasso
plt.plot(alphas, scores,linestyle='dashed',marker='o',markerfacecolor='blue',markersize=3)
plt.xlabel('alpha')
plt.ylabel('score')
plt.grid(True)
plt.title('Alphas vs. Score')
plt.show()
print "The Optimal Alpha and Score of Lasso Linear is: ", opt_a, max_score
